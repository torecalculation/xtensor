512 template <class EC, std::size_t N, layout_type L, class Tag>
513 template <class SC>
514 inline xtensor_container<EC, N, L, Tag>& xtensor_container<EC, N, L, Tag>::operator=(xarray_container<EC, L, SC, Tag>&& rhs)
515 {
516 XTENSOR_ASSERT_MSG(N == rhs.dimension(), "Cannot change dimension of xtensor.");
517 std::copy(rhs.shape().begin(), rhs.shape().end(), this->shape_impl().begin());
518 std::copy(rhs.strides().cbegin(), rhs.strides().cend(), this->strides_impl().begin());
519 std::copy(rhs.backstrides().cbegin(), rhs.backstrides().cend(), this->backstrides_impl().begin());
520 this->mutable_layout() = std::move(rhs.layout());
521 m_storage = std::move(std::move(rhs.storage()));
522 return *this;
523 }
